🗃 Migration: Package ↔ Course Links

One-time setup to enable linking packages to their applicable courses.

📋 What this creates
→A new junction table package_courses (package_id + course_id) with a unique constraint so no duplicate links can exist.
→Row-Level Security enabled with a policy so authenticated admins can read and write course links.
→Indexes for fast lookup by package or by course.
✅ Safe to run multiple times — uses CREATE TABLE IF NOT EXISTS and CREATE POLICY IF NOT EXISTS.

-- ════════════════════════════════════════════════════════════
-- MIGRATION: package_courses junction table
-- Minds' Craft — links packages to the courses they apply to
-- Safe to run multiple times (idempotent)
-- ════════════════════════════════════════════════════════════

-- 1. Create the junction table
CREATE TABLE IF NOT EXISTS package_courses (
  id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  package_id  UUID NOT NULL REFERENCES packages(id)  ON DELETE CASCADE,
  course_id   UUID NOT NULL REFERENCES courses(id)   ON DELETE CASCADE,
  created_at  TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE (package_id, course_id)   -- no duplicate links
);

-- 2. Indexes for fast lookups
CREATE INDEX IF NOT EXISTS idx_pkg_courses_package ON package_courses(package_id);
CREATE INDEX IF NOT EXISTS idx_pkg_courses_course  ON package_courses(course_id);

-- 3. Enable Row-Level Security
ALTER TABLE package_courses ENABLE ROW LEVEL SECURITY;

-- 4. Policy: authenticated users have full access
DO $$
BEGIN
  IF NOT EXISTS (
    SELECT 1 FROM pg_policies
    WHERE tablename = 'package_courses'
      AND policyname = 'authenticated_full_access'
  ) THEN
    CREATE POLICY "authenticated_full_access" ON package_courses
      FOR ALL
      USING  (auth.uid() IS NOT NULL)
      WITH CHECK (auth.uid() IS NOT NULL);
  END IF;
END $$;

-- 5. Verify
SELECT 'package_courses table ready ✅' AS status;
SELECT column_name, data_type
FROM information_schema.columns
WHERE table_name = 'package_courses'
ORDER BY ordinal_position;
